public class hello {
    public static void main(String[] args){
        //How to print
        System.out.println("Hello World");

        //Variables
        int num = 5;
        double doubley = 4.6;
        char firstInitial = 'J';
        String name = "Justin";
        boolean isHome = true;


        System.out.println(num);
        System.out.println(doubley);
        System.out.println(name);
        System.out.println(firstInitial);
        System.out.println(isHome);

        if (num<18){
            System.out.println("You are not an Adult yet...");
        }

        if (!isHome && num < 18){
            System.out.println("You are not an Adult yet...");
        }

        if (name.equals("Justin")){
            System.out.println("Your Learning Java");

        }else {
            System.out.println("Have FUN!!!");
        }
        
        while (num<18){
            System.out.println("Your not an adult");
            //num = num + 1
            num++;
        }

        
    }
}
